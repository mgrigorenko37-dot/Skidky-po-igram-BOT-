import threading
import time
import logging
import sqlite3
from bot_init import bot 

# Импортируем настройки и базу
import config
import database
import utils

# Импортируем системные модули
import menu_system
import games_system
import mining_system
import wallet_system

# Настройка логгера
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# === ФОНОВЫЕ ЗАДАЧИ ===
def background_tasks():
    print("🔄 Фоновые задачи запущены...")
    # Запуск циклов отслеживания цен и новостей
    games_system.start_background_tasks()
    
    while True:
        try:
            # Здесь можно добавить логику проверки поломок ферм раз в час
            time.sleep(3600)
        except Exception as e:
            logger.error(f"Background task error: {e}")
            time.sleep(600)

# === ЗАПУСК СИСТЕМ ===
if __name__ == "__main__":
    print("🛠 Инициализация базы данных...")
    database.init_db()

    print("🔌 Активация обработчиков команд...")
    # РЕГИСТРАЦИЯ МОДУЛЕЙ (Оживляем кнопки)
    menu_system.register_handlers(bot)
    games_system.register_handlers(bot)
    mining_system.register_handlers(bot)
    wallet_system.register_handlers(bot)

    # Базовая команда /start
    @bot.message_handler(commands=['start'])
    def start_handler(message):
        ref_id = None
        if len(message.text.split()) > 1:
            try: ref_id = int(message.text.split()[1])
            except: pass
        
        # Регистрация пользователя в БД
        with sqlite3.connect(config.DB_FILE) as conn:
            conn.execute("INSERT OR IGNORE INTO user_news_prefs (user_id, referrer_id) VALUES (?, ?)", 
                         (message.chat.id, ref_id))
        
        # Отправка приветствия с главной клавиатурой
        utils.safe_send_message(
            message.chat.id, 
            f"👋 Привет, {message.from_user.first_name}!\nЯ твой игровой брокер и менеджер фермы.", 
            reply_markup=utils.main_kbrd(message.chat.id), 
            parse_mode='Markdown'
        )

    # Запуск фонового потока
    threading.Thread(target=background_tasks, daemon=True).start()

    print("🚀 Бот Game Broker успешно запущен!")
    try:
        bot.infinity_polling(skip_pending=True)
    except Exception as e:
        logger.error(f"Critical Bot Error: {e}")
