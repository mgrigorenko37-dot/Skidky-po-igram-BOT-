# --- [АНТИ-ФЛУД] Защита от быстрых нажатий (debounce) ---
COOLDOWNS = {}  # Формат: {user_id: {action: timestamp}}
FLOOD_TIMEOUT = 2  # Секунды между одинаковыми действиями

def check_cooldown(user_id, action, timeout=FLOOD_TIMEOUT):
    """
    Проверяет, прошло ли достаточно времени с последнего действия.
    Возвращает True если можно выполнять, False если нужно подождать.
    """
    current_time = time.time()
    if user_id not in COOLDOWNS:
        COOLDOWNS[user_id] = {}
    
    last_time = COOLDOWNS[user_id].get(action, 0)
    time_passed = current_time - last_time
    
    if time_passed >= timeout:
        COOLDOWNS[user_id][action] = current_time
        return True
    return False

def setup_logging():
    """Настройка логирования с ротацией файлов"""
    log_format = '%(asctime)s - %(name)s - %(levelname)s - %(message)s'

    try:
        # Файловый лог (макс 10МБ, 5 файлов ротации)
        file_handler = RotatingFileHandler(
            'bot.log', 
            maxBytes=10*1024*1024, 
            backupCount=5,
            encoding='utf-8'
        )
        file_handler.setLevel(logging.INFO)
        file_handler.setFormatter(logging.Formatter(log_format))

        # Консольный лог
        console_handler = logging.StreamHandler()
        console_handler.setLevel(logging.WARNING)
        console_handler.setFormatter(logging.Formatter(log_format))

        # Настройка корневого логгера
        logging.basicConfig(
            level=logging.INFO,
            handlers=[file_handler, console_handler],
            format=log_format
        )

        return logging.getLogger(__name__)
    except Exception as e:
        # Если не удалось настроить файловое логирование, используем только консоль
        logging.basicConfig(level=logging.INFO, format=log_format)
        logger = logging.getLogger(__name__)
        logger.warning(f"File logging failed, using console only: {e}")
        return logger

# --- [БЕЗОПАСНОСТЬ] Умная обертка для Telegram API ---
def safe_api_call(func, *args, **kwargs):
    """
    Безопасный вызов Telegram API методов.
    Возвращает результат функции или None при ошибках, которые можно проигнорировать.
    """
    try:
        return func(*args, **kwargs)
    except telebot.apihelper.ApiTelegramException as e:
        error_msg = str(e).lower()

        # Категории ошибок, которые можно безопасно игнорировать
        ignorable_errors = [
            "message to edit not found",
            "message is not modified",
            "message to delete not found",
            "chat not found",
            "bot was blocked by the user",
            "user is deactivated",
            "not enough rights",
            "message can't be deleted",
            "message identifier is not specified"
        ]

        if any(err in error_msg for err in ignorable_errors):
            # Тихий логинг только для блокировок
            if "bot was blocked" in error_msg:
                logger.info(f"User blocked the bot")
            return None
        else:
            # Критические ошибки логируем
            logger.error(f"Telegram API Error [{func.__name__}]: {e}")
            return None
    except Exception as e:
        logger.error(f"Unexpected error in {func.__name__}: {e}")
        return None

# --- [УТИЛИТЫ] Безопасная работа с JSON ---
def safe_json_loads(json_str, default=None):
    """Безопасная загрузка JSON с обработкой ошибок"""
    if default is None:
        default = {}

    if not json_str or not isinstance(json_str, str):
        return default

    try:
        return json.loads(json_str)
    except (json.JSONDecodeError, TypeError, ValueError) as e:
        logger.warning(f"JSON decode error: {e}, data: {json_str[:100]}")
        return default

def safe_json_dumps(data, default=None):
    """Безопасная сериализация в JSON"""
    if default is None:
        default = "{}"

    try:
        return json.dumps(data, ensure_ascii=False)
    except (TypeError, ValueError) as e:
        logger.error(f"JSON encode error: {e}")
        return default

BUFFER_CAPACITY_RATIO = 2.0  # Вместимость буфера = доход за 2 дня

def calculate_farm_income(user_id):
    # Считаем доход по каждой установленной детали
    # Учитываем: Тип детали (Part Ratio), Уровень (Tier Income), Поломку, Старение
    total_income = 0.0
    with sqlite3.connect(DB_FILE) as conn:
        conn.row_factory = sqlite3.Row
        # Получаем все детали в ферме, джойним с инвентарем чтобы знать уровень (tier)
        query = """
            SELECT f.slot_name, f.is_broken, f.restores_count, i.tier, i.rarity 
            FROM mining_farm f
            JOIN mining_inventory i ON f.inventory_item_id = i.id
            WHERE f.user_id = ?
        """
        parts = conn.execute(query, (user_id,)).fetchall()

        for p in parts:
            if p['is_broken']: continue # Сломанная деталь не майнит

            # Базовый доход уровня (например, Элита = $11.60)
            tier_info = TIERS_CONFIG.get(p['tier'], TIERS_CONFIG[1])
            base_tier_income = tier_info['income']

            # Доля дохода этой детали (GPU = 25%, Mouse = 2.5%)
            ratio = PART_INCOME_RATIO.get(p['rarity'], 0.0)

            # Штраф за старение (каждая реставрация -10%)
            aging_factor = 1.0 - (p['restores_count'] * AGING_PENALTY)
            if aging_factor < 0.1: aging_factor = 0.1 # Минимум 10% эффективности

            # ИТОГ ПО ДЕТАЛИ
            part_income = base_tier_income * ratio * aging_factor
            total_income += part_income

    return round(total_income, 4)

def update_mining_buffer(user_id):
    """Обновляет накопления в буфере на основе прошедшего времени"""
    try:
        with sqlite3.connect(DB_FILE) as conn:
            cursor = conn.cursor()
            cursor.execute("SELECT buffer_usdt, last_collect_date FROM mining_users WHERE user_id = ?", (user_id,))
            row = cursor.fetchone()
            
            # Если юзера нет в таблице майнинга - создаем его
            if not row:
                cursor.execute("INSERT INTO mining_users (user_id, reg_date, balance_usdt, buffer_usdt) VALUES (?, ?, 0.0, 0.0)", 
                             (user_id, datetime.now().strftime('%Y-%m-%d')))
                conn.commit()
                buffer, last_date = 0.0, None
            else:
                buffer, last_date = row
            
            now = datetime.now()
            income_per_day = calculate_farm_income(user_id)
            
            # Лимит хранилища (буфера) - не более дохода за 24 часа (100% от дневного дохода)
            # Пользователь должен заходить хотя бы раз в сутки, иначе майнинг "простаивает"
            max_buffer = round(income_per_day, 6)
            if max_buffer < 0.001: max_buffer = 0.001 # Минимальный лимит для совсем слабых ферм
            
            if not last_date:
                cursor.execute("UPDATE mining_users SET last_collect_date = ? WHERE user_id = ?", (str(now), user_id))
                conn.commit()
                return buffer, max_buffer, 0
                
            last_dt = datetime.fromisoformat(last_date)
            seconds_passed = (now - last_dt).total_seconds()
            
            # Расчет дохода: (Доход_в_день / 86400) * секунды
            income_per_second = income_per_day / 86400.0
            new_accrual = seconds_passed * income_per_second
            
            # Итоговый буфер с жестким ограничением
            final_buffer = min(buffer + new_accrual, max_buffer)
            
            # Обновляем базу
            cursor.execute("UPDATE mining_users SET buffer_usdt = ?, last_collect_date = ? WHERE user_id = ?", 
                         (final_buffer, str(now), user_id))
            conn.commit()
            
            percent = int((final_buffer / max_buffer * 100)) if max_buffer > 0 else 0
            if percent > 100: percent = 100
            
            return round(final_buffer, 6), max_buffer, percent
    except Exception as e:
        logger.error(f"Buffer update error: {e}")
        return 0, 0, 0

def open_box_rng(tier):
    # Гача-рандом на основе конфига DROP_RATES
    rates = DROP_RATES.get(tier, DROP_RATES[2]) # По дефолту Любитель
    roll = random.uniform(0, 100)

    # 1. Проверяем Ядро (Самое редкое)
    if roll < rates['core']:
        rarity = "core"
        part_type = random.choice(PC_STRUCTURE["core"])
    # 2. Проверяем Железо
    elif roll < (rates['core'] + rates['iron']):
        rarity = "iron"
        part_type = random.choice(PC_STRUCTURE["iron"])
    # 3. Иначе Мусор
    else:
        rarity = "trash"
        part_type = random.choice(PC_STRUCTURE["trash"])

    return part_type, rarity

def trigger_daily_breakdown(user_id):
    # Умная поломка: проверяем иммунитет и кошелек
    with sqlite3.connect(DB_FILE) as conn:
        conn.row_factory = sqlite3.Row
        user = conn.execute("SELECT reg_date, balance_usdt FROM mining_users WHERE user_id = ?", (user_id,)).fetchone()
        if not user: return None

        # 1. Иммунитет новичка
        reg_date = datetime.strptime(user['reg_date'], '%Y-%m-%d')
        if (datetime.now() - reg_date).days < BREAKDOWN_IMMUNITY_DAYS:
            return None # Не ломаем

        # 2. Правило "Тормоз" (не ломаем, если ремонт стоит > 50% дохода)
        daily_income = calculate_farm_income(user_id)
        if daily_income <= 0: return None

        # Выбираем деталь для поломки
        installed = conn.execute("SELECT slot_name, inventory_item_id FROM mining_farm WHERE user_id = ? AND is_broken = 0", (user_id,)).fetchall()
        if not installed: return None

        target = random.choice(installed)

        # Тут должна быть проверка: есть ли деньги на ремонт этой детали?
        # Для упрощения пока просто ломаем рандомную рабочую деталь
        conn.execute("UPDATE mining_farm SET is_broken = 1 WHERE user_id = ? AND slot_name = ?", (user_id, target['slot_name']))
        conn.commit()
        logger.info(f"Part breakdown: user={user_id}, part={target['slot_name']}")
        return target['slot_name']

def get_game_best_deal(game_name, allowed_stores=None):
    try:
        url = f"https://www.cheapshark.com/api/1.0/games?title={game_name}&limit=5"

        try:
            response = requests.get(url, timeout=15)
            if response.status_code != 200:
                logger.error(f"CheapShark API error: {response.status_code}")
                return None

            data = response.json()
        except requests.exceptions.Timeout:
            logger.error(f"Timeout error for game: {game_name}")
            return None
        except requests.exceptions.RequestException as e:
            logger.error(f"Request error for game {game_name}: {e}")
            return None

        if not data:
            return None

        # Получаем первую игру из результатов
        game = data[0]
        game_id = game['gameID']
        thumb = game['thumb']

        # Получаем детальную информацию об игре
        detail_url = f"https://www.cheapshark.com/api/1.0/games?id={game_id}"
        try:
            detail_response = requests.get(detail_url, timeout=15)
            if detail_response.status_code != 200:
                return None
            full_data = detail_response.json()
        except Exception as e:
            logger.error(f"Detail API error for game {game_id}: {e}")
            return None

        deals = full_data.get('deals', [])

        # Фильтруем по разрешенным магазинам
        if allowed_stores:
            filtered_deals = []
            for deal in deals:
                if deal['storeID'] in allowed_stores:
                    filtered_deals.append(deal)
            deals = filtered_deals

        if not deals:
            return None

        # Находим лучшую сделку (самую низкую цену)
        best = min(deals, key=lambda x: float(x['price']))

        # Формируем сырую ссылку
        raw_url = f"https://www.cheapshark.com/redirect?dealID={best['dealID']}"

        # Генерируем партнерскую ссылку
        aff_url = generate_aff_link(raw_url, best['storeID'])
        if not aff_url:
            aff_url = raw_url  # Фолбэк

        # Получаем полную ссылку на изображение
        if thumb and not thumb.startswith('http'):
            thumb = 'https://www.cheapshark.com' + thumb

        return {
            "title": game['external'],
            "price": best['price'],
            "url": aff_url,
            "thumb": thumb,
            "storeID": best['storeID']
        }

    except Exception as e:
        logger.error(f"Error getting game deal for '{game_name}': {e}")
        return None

RATES_CACHE = {"rates": {}, "time": 0}

def get_rates():
    if time.time() - RATES_CACHE["time"] > 3600 or not RATES_CACHE["rates"]: 
        try:
            resp = requests.get("https://open.er-api.com/v6/latest/USD", timeout=10).json()
            if resp.get('result') == 'success':
                RATES_CACHE["rates"] = resp['rates']
                RATES_CACHE["time"] = time.time()
                logger.info("Exchange rates updated")
        except Exception as e:
            logger.error(f"Exchange rates API error: {e}")
            if not RATES_CACHE["rates"]:
                RATES_CACHE["rates"] = {'USD': 1.0, 'RUB': 90.0, 'KZT': 450.0, 'UAH': 38.0}
    return RATES_CACHE["rates"]

def format_price(usd_price, user_id=None):
    if not usd_price: return ""
    try:
        price_float = float(usd_price)
        if price_float == 0: return "Бесплатно"
        target_curr = 'USD'
        if user_id:
            target_curr = get_user_currency(user_id)
        rates = get_rates()
        if target_curr not in rates:
            target_curr = 'USD'
        final_price = price_float * rates[target_curr]
        sign = CURRENCY_SIGNS.get(target_curr, target_curr)
        if target_curr == 'USD':
            return "" 
        return f"~{int(final_price) if final_price > 10 else round(final_price, 2)} {sign}"
    except: return ""

def get_image_from_url(url):
    try:
        r = requests.get(url, headers={'User-Agent': 'Mozilla/5.0'}, timeout=10)
        soup = BeautifulSoup(r.text, 'html.parser')
        og_image = soup.find('meta', property='og:image')
        if isinstance(og_image, telebot.types.Tag) and og_image.get('content'):
            return str(og_image['content'])
        for img in soup.find_all('img'):
            src = img.get('src')
            if src and isinstance(src, str) and (src.startswith('http') or src.startswith('//')):
                if 'logo' not in src.lower(): return src if src.startswith('http') else 'https:' + src
        return None
    except Exception as e:
        logger.error(f"Error getting image from URL: {e}")
        return None

            pc_build = safe_json_loads(row[1] if row and row[1] else "{}")
            balance = row[2] if row else 0

            if inventory.get(key_to_craft, 0) < 5:
                safe_answer_callback(c.id, "Нужно 5 предметов!", show_alert=True)
                return

            is_success = False

            if action == "safe":
                price = safe_prices.get(p_rarity, 50)
                if balance < price:
                    safe_answer_callback(c.id, "Недостаточно Stars!", show_alert=True)
                    return
                balance -= price
                is_success = True

            elif action == "risk":
                rate = success_rates.get(p_rarity, 50)
                if random.randint(1, 100) <= rate:
                    is_success = True
                else:
                    is_success = False

            inventory[key_to_craft] -= 5
            if inventory[key_to_craft] == 0:
                del inventory[key_to_craft]

            msg_res = ""
            if is_success:
                new_rarity = p_rarity + 1
                candidates = [p for p in PC_PARTS[p_type] if p['rarity'] == new_rarity]
                if not candidates: candidates = [p for p in PC_PARTS[p_type] if p['rarity'] >= new_rarity]
                new_part = random.choice(candidates)

                old_part = pc_build.get(p_type, {"score": 0, "name": "Пустой слот"})

                msg_res = f"🔨 **КРАФТ УСПЕШЕН!**\n\nПолучено: **{new_part['name']}**\n"

                if new_part['score'] > old_part['score']:
                    if old_part['score'] > 0:
                        old_key = f"{p_type}_{old_part['rarity']}"
                        inventory[old_key] = inventory.get(old_key, 0) + 1
                        msg_res += f"✅ **АПГРЕЙД!** Установлена в ПК. Старая ({old_part['name']}) -> в инвентарь."
                    else:
                        msg_res += "✅ **Установлена в ПК!**"
                    pc_build[p_type] = new_part
                else:
                    new_key = f"{p_type}_{new_rarity}"
                    inventory[new_key] = inventory.get(new_key, 0) + 1
                    msg_res += f"📦 Новая деталь отправлена в инвентарь."
            else:
                msg_res = "💥 **НЕУДАЧА...**\n\nДетали сгорели при попытке крафта. Попробуйте использовать страховку."

            cursor.execute("UPDATE user_news_prefs SET parts_inventory = ?, pc_data = ?, balance = ? WHERE user_id = ?", 
                           (safe_json_dumps(inventory), safe_json_dumps(pc_build), balance, user_id))
            conn.commit()

        safe_send_message(user_id, msg_res, parse_mode='Markdown')
        crafting_menu(c.message)
    except Exception as e:
        logger.error(f"Crafting error for user {user_id}: {e}")

# --- [ОПТИМИЗИРОВАНО] Безопасная анимация лутбокса ---
def process_lootbox_animation(user_id, level):
    try:
        # Используем safe_api_call для защиты от падений
        msg = safe_send_message(user_id, "📦 **Заказ оформлен...**", parse_mode='Markdown')
        if not msg:
            # Если не удалось отправить сообщение, сразу переходим к логике
            process_loot_logic(user_id, level)
            return

        frames = ["🚚 **В пути...** 🚛", "🏭 **На складе...** 🏭", "🔧 **Проверка качества...** 🔍", "📦 **Распаковка...** ✂️", "✨ **Готово!**"]

        for frame in frames:
            time.sleep(0.7)
            # Если сообщение было удалено юзером, бот не упадет
            safe_edit_message_text(frame, user_id, msg.message_id, parse_mode='Markdown')

        # Удаляем сервисное сообщение анимации
        safe_delete_message(user_id, msg.message_id)

        # Запускаем логику выдачи предмета
        process_loot_logic(user_id, level)
    except Exception as e:
        logger.error(f"Critical error in animation for user {user_id}: {e}")
        # Даже если анимация упала, выдаем предмет
        process_loot_logic(user_id, level)

# --- [ОПТИМИЗИРОВАНО] Безопасная логика выдачи лутбокса ---
def process_loot_logic(user_id, level):
    try:
        min_rarity_map = {1: 1, 2: 2, 3: 2, 4: 3, 5: 4}
        min_rarity = min_rarity_map.get(level, 1)

        bonus_chance = {1: 5, 2: 15, 3: 30, 4: 50, 5: 80}
        chance = bonus_chance.get(level, 0)

        roll = random.randint(1, 100)
        target_rarity = min_rarity
        if roll <= chance:
            target_rarity += 1
            if target_rarity > 5: target_rarity = 5

        part_type = random.choice(list(PC_PARTS.keys()))

        candidates = [p for p in PC_PARTS[part_type] if p['rarity'] >= target_rarity]

        if not candidates:
            candidates = [p for p in PC_PARTS[part_type] if p['rarity'] >= min_rarity]
            if not candidates:
                 candidates = PC_PARTS[part_type]

        if not candidates:
            logger.error(f"No candidates for part_type {part_type}")
            safe_send_message(user_id, "❌ Ошибка при открытии коробки. Попробуйте еще раз.")
            return

        new_part = random.choice(candidates)
        part_img = PART_IMAGES.get(part_type, PART_IMAGES['CASE'])

        with sqlite3.connect(DB_FILE) as conn:
            cursor = conn.cursor()
            row = cursor.execute("SELECT pc_data, balance, parts_inventory FROM user_news_prefs WHERE user_id = ?", (user_id,)).fetchone()

            # Используем безопасную загрузку JSON
            pc_build = safe_json_loads(row[0] if row and row[0] else "{}")
            balance = row[1] if row else 0
            inventory = safe_json_loads(row[2] if row and row[2] else "{}")

            old_part = pc_build.get(part_type, {"score": 0, "name": "Пустой слот", "price": 0})

            type_names = {
                "CASE": "Корпус", "MONITOR": "Монитор", "MOUSE": "Мышь", 
                "PAD": "Коврик", "CPU": "Процессор", "GPU": "Видеокарта", 
                "RAM": "ОЗУ", "DISK": "Диск"
            }
            type_ru = type_names.get(part_type, part_type)

            rarity_stars = "⭐" * new_part['rarity']
            msg_header = f"🎁 **Коробка (Ур. {level}) открыта!**\n\n{rarity_stars}\n🔧 **{type_ru}: {new_part['name']}**\n⚡ Мощность: {new_part['score']}\n💰 Цена: {new_part['price']} Stars\n\n"

            if new_part['score'] > old_part['score']:
                pc_build[part_type] = new_part
                action_text = f"✅ **АПГРЕЙД!**\nСтарая деталь ({old_part['name']}) была хуже.\nМощность вашего ПК выросла!"

                if old_part['score'] > 0:
                    old_key = f"{part_type}_{old_part['rarity']}"
                    inventory[old_key] = inventory.get(old_key, 0) + 1
                    action_text += f"\n📦 Старая деталь ({old_part['name']}) отправлена в инвентарь."
            else:
                new_key = f"{part_type}_{new_part['rarity']}"
                inventory[new_key] = inventory.get(new_key, 0) + 1
                action_text = f"📦 **В инвентарь.**\nТекущая деталь мощнее. Эта отправлена на склад (для крафта)."

            cursor.execute("UPDATE user_news_prefs SET pc_data = ?, parts_inventory = ? WHERE user_id = ?", 
                           (safe_json_dumps(pc_build), safe_json_dumps(inventory), user_id))
            conn.commit()

            try:
                safe_send_message(user_id, part_img, caption=msg_header + action_text, parse_mode='Markdown')
            except Exception as e:
                logger.error(f"Error sending photo to {user_id}: {e}")
                safe_send_message(user_id, msg_header + action_text, parse_mode='Markdown')

    except Exception as e:
        logger.error(f"Error in loot logic for user {user_id}, level {level}: {e}")
        safe_send_message(user_id, "❌ Произошла ошибка при открытии коробки. Обратитесь к администратору.")

def send_genre_deals(chat_id, genre_tag):
    try:
        # Сначала отправляем сообщение о поиске
        search_msg = safe_send_message(chat_id, f"🔍 Ищу игры в жанре {GENRES.get(genre_tag, genre_tag)}...")

        # Ищем игры по жанру
        random_page = random.randint(0, 50) # Увеличиваем диапазон страниц для разнообразия

        # Ищем игры со скидками
        url = f"https://www.cheapshark.com/api/1.0/deals?storeID=1,7,25&pageSize=60&pageNumber={random_page}&sortBy=Recent" # Сортируем по новизне для разнообразия

        try:
            response = requests.get(url, timeout=15)
            if response.status_code != 200:
                if search_msg: safe_delete_message(chat_id, search_msg.message_id)
                safe_send_message(chat_id, "😔 Сервер скидок временно недоступен.")
                return
            all_deals = response.json()
        except Exception as e:
            logger.error(f"API request error for genre {genre_tag}: {e}")
            if search_msg: safe_delete_message(chat_id, search_msg.message_id)
            safe_send_message(chat_id, "😔 Ошибка при загрузке игр.")
            return

        if search_msg: safe_delete_message(chat_id, search_msg.message_id)

        if not all_deals:

        safe_send_message(m.chat.id, "⚙️ **Настройки уведомлений и магазинов:**",

        support_info(m)
        return

        admin_stat(m)

        return

        name_en = GoogleTranslator(source='auto', target='en').translate(m.text)
    except Exception as e:
        logger.error(f"Translation error for search '{m.text}': {e}")
        safe_send_message(m.chat.id, "❌ Ошибка перевода. Попробуйте ввести название на английском.")

            safe_send_message(m.chat.id, "❌ Ошибка сервера поиска. Попробуйте позже.")
            return

        if not all_games:
            safe_send_message(m.chat.id, "🔍 Ничего не найдено. Попробуйте точное название на английском.")
            return

        # Дедупликация по названию
        seen_names = set()
        unique_games = []
        for g in all_games:
            name_lower = g['external'].lower()
            if name_lower not in seen_names:
                seen_names.add(name_lower)
                unique_games.append(g)
        
        # Берем топ 5 уникальных
        games = unique_games[:5]

        mk = types.InlineKeyboardMarkup(row_width=1)
        for game in games:
            mk.add(types.InlineKeyboardButton(f"{game['external']}", callback_data=f"find_{game['gameID']}"))

    except Exception as e:
        logger.error(f"Search error for '{m.text}': {e}")
        safe_send_message(m.chat.id, "❌ Ошибка при поиске.")

            game_id = c.data.split('_')[1]
            url = f"https://www.cheapshark.com/api/1.0/games?id={game_id}"
            data = requests.get(url).json()
            title = data['info']['title']
            price = float(data['deals'][0]['price']) if data.get('deals') else 0.0

            with sqlite3.connect(DB_FILE) as conn:

                             (c.message.chat.id, title, price))

            safe_delete_message(c.message.chat.id, c.message.message_id)

        elif c.data.startswith('tgl_'):
            col_name = c.data.replace('tgl_', '')
            
            # Сопоставление callback_data с колонками БД
            col_map = {
                'news': 'want_news',
                'free': 'want_freebies'
            }

            if col_name not in col_map:
                logger.warning(f"Unknown toggle: {c.data}")
                safe_answer_callback(c.id, "❌ Ошибка")
                return

            safe_col = col_map[col_name]
            with sqlite3.connect(DB_FILE) as conn:
                conn.execute(
                    f"UPDATE user_news_prefs SET {safe_col} = 1 - {safe_col} WHERE user_id = ?",
                    (c.from_user.id,)
                )
                conn.commit()

            safe_edit_message_reply_markup(c.message.chat.id, c.message.message_id, reply_markup=settings_kbrd(c.from_user.id))
            safe_answer_callback(c.id, "✅ Настройки обновлены")

        elif c.data == "change_currency":
            safe_edit_message_text("💱 **Выберите валюту:**", c.message.chat.id, c.message.message_id, reply_markup=currency_kbrd(), parse_mode='Markdown')

        elif c.data.startswith("set_curr_"):
            curr = c.data.split('_')[2]

                conn.execute("UPDATE user_news_prefs SET currency = ? WHERE user_id = ?", (curr, c.from_user.id))

        elif c.data == "manage_stores":
            u_s = get_user_stores(c.from_user.id)
            mk = types.InlineKeyboardMarkup(row_width=2).add(*[types.InlineKeyboardButton(f"{'✅' if i in u_s else '❌'} {n}", callback_data=f"st_{i}") for i, n in SUPPORTED_STORES.items()])
            mk.add(types.InlineKeyboardButton("🔙 Назад", callback_data="back_to_settings")) 
            safe_edit_message_text("Магазины:", c.message.chat.id, c.message.message_id, reply_markup=mk)
        elif c.data.startswith('st_'):
            sid = c.data.split('_')[1]
            with sqlite3.connect(DB_FILE) as conn:
                cursor = conn.cursor()
                cursor.execute("SELECT 1 FROM user_stores WHERE user_id = ? AND store_id = ?", (c.from_user.id, sid))
                if cursor.fetchone(): conn.execute("DELETE FROM user_stores WHERE user_id = ? AND store_id = ?", (c.from_user.id, sid))
                else: conn.execute("INSERT INTO user_stores (user_id, store_id) VALUES (?, ?)", (c.from_user.id, sid))
            u_s = get_user_stores(c.from_user.id)
            mk = types.InlineKeyboardMarkup(row_width=2).add(*[types.InlineKeyboardButton(f"{'✅' if i in u_s else '❌'} {n}", callback_data=f"st_{i}") for i, n in SUPPORTED_STORES.items()])
            mk.add(types.InlineKeyboardButton("🔙 Назад", callback_data="back_to_settings"))
            safe_edit_message_reply_markup(c.message.chat.id, c.message.message_id, reply_markup=mk)
        elif c.data.startswith('del_'):
            gp = c.data.split('_')[1].lower()
            with sqlite3.connect(DB_FILE) as conn:
                conn.execute("DELETE FROM wishlist WHERE user_id = ? AND game_name LIKE ?", (c.from_user.id, gp + '%'))
            safe_delete_message(c.message.chat.id, c.message.message_id)
        safe_answer_callback(c.id)
    except Exception as e:
        logger.error(f"Callback handler error: {e}")

def price_tracker_loop():
    while True:
        try:
            with sqlite3.connect(DB_FILE) as conn:
                cursor = conn.cursor()
                cursor.execute("SELECT user_id, game_name, last_price FROM wishlist")
                items = cursor.fetchall()
                for uid, g_name, old_price in items:
                    current_deal = get_game_best_deal(g_name, get_user_stores(uid))
                    if current_deal:
                        new_price = float(current_deal['price'])
                        if new_price < old_price:
                            store_name = SUPPORTED_STORES.get(current_deal['storeID'], "магазине")

                            conv_price = format_price(new_price, uid)
                            price_text = f"**${new_price}**"
                            if conv_price: price_text += f" ({conv_price})"

                            msg_text = (f"📉 **Цена упала!**\n\n"
                                        f"Игра: **{current_deal['title']}**\n"
                                        f"Старая цена: `${old_price}`\n"
                                        f"Новая цена: {price_text}\n"
                                        f"Где: {store_name}")
                            mk = types.InlineKeyboardMarkup()
                            mk.add(types.InlineKeyboardButton("🛒 Купить сейчас", url=current_deal['url']))
                            mk.add(types.InlineKeyboardButton("🟠 Найти на GGSEL", url=get_ggsel_link(current_deal['title'])))

                            try:
                                safe_send_message(uid, current_deal['thumb'], caption=msg_text, parse_mode='Markdown', reply_markup=mk)
                                cursor.execute("UPDATE wishlist SET last_price = ? WHERE user_id = ? AND game_name = ?", (new_price, uid, g_name))
                            except Exception as e:
                                logger.warning(f"Failed to send price drop to {uid}: {e}")
                conn.commit()
        except Exception as e:
            logger.error(f"❌ Ошибка трекера цен: {e}")
        time.sleep(3600)

def auto_sales_scanner():
    try:
        url = "https://www.cheapshark.com/api/1.0/deals?storeID=1&metacritic=75&steamRating=80&onSale=1&pageSize=20&sortBy=Savings"
        deals = requests.get(url, timeout=20).json()

        with sqlite3.connect(DB_FILE) as conn:
            cursor = conn.cursor()
            sent_count = 0

            for deal in deals:
                if float(deal['savings']) < 10: continue
                if float(deal['normalPrice']) < 5: continue

                game_id = deal['gameID']

                cursor.execute("SELECT 1 FROM posted_deals WHERE game_id = ?", (game_id,))
                if cursor.fetchone(): continue

                aff_url = generate_aff_link(f"https://www.cheapshark.com/redirect?dealID={deal['dealID']}", "1")
                discount = int(float(deal['savings']))

                caption = (f"🔥 **ГОРЯЧАЯ СКИДКА!**\n\n"
                           f"🎮 **{deal['title']}**\n"
                           f"📉 Скидка: **-{discount}%**\n"
                           f"💰 Цена: **${deal['salePrice']}** (было ${deal['normalPrice']})\n"
                           f"⭐️ Steam: {deal['steamRatingPercent']}%\n"
                           f"🏆 Metacritic: {deal['metacriticScore']}")

                mk = types.InlineKeyboardMarkup()
                mk.add(types.InlineKeyboardButton("🛒 Купить в Steam", url=aff_url))
                mk.add(types.InlineKeyboardButton("🟠 Найти на GGSEL", url=get_ggsel_link(deal['title'])))

                broadcast_news(deal['title'], aff_url, "sale", deal['thumb'], caption, mk)

                cursor.execute("INSERT OR REPLACE INTO posted_deals (game_id, date_sent) VALUES (?, ?)", (game_id, str(datetime.now())))
                conn.commit()

                sent_count += 1
                break 

    except Exception as e:
        logger.error(f"Sales scanner error: {e}")

def process_rss_loop():
    while True:
        try:
            process_rss(RSS_FREEBIES, "freebie")
            process_rss(RSS_NEWS, "news")
            auto_sales_scanner()
            time.sleep(3600) 
        except Exception as e:
            logger.error(f"RSS loop error: {e}")
            time.sleep(60)

def process_rss(url, type_news):
    try:
        feed = feedparser.parse(url)
        max_check = 5 
        with sqlite3.connect(DB_FILE) as conn:
            cursor = conn.cursor()
            for entry in feed.entries[:max_check]:
                cursor.execute("SELECT 1 FROM sent_news WHERE news_id = ?", (entry.link,))
                if not cursor.fetchone():
                    img = extract_image(entry)
                    broadcast_news(entry.title, entry.link, type_news, img)
                    cursor.execute("INSERT INTO sent_news (news_id) VALUES (?)", (entry.link,))
            conn.commit()
    except Exception as e:
        logger.error(f"❌ Ошибка RSS ({url}): {e}")

@bot.message_handler(commands=['convert'])
def convert_currency_cmd(message):
    try:
        parts = message.text.split()
        if len(parts) != 4:
            safe_send_message(message.chat.id, "ℹ️ Используйте: /convert <сумма> <из_валюты> <в_валюту>\nПример: `/convert 100 USD RUB`", parse_mode='Markdown')
            return

        amount = float(parts[1])
        base = parts[2].upper()
        target = parts[3].upper()

        rates = get_rates()

        # Конвертация через USD (базовая в API)
        usd_amount = amount
        if base != 'USD':
            if base not in rates:
                safe_send_message(message.chat.id, f"❌ Неизвестная валюта: {base}")
                return
            usd_amount = amount / rates[base]

        final_amount = usd_amount
        if target != 'USD':
            if target not in rates:
                safe_send_message(message.chat.id, f"❌ Неизвестная валюта: {target}")
                return
            final_amount = usd_amount * rates[target]

        safe_send_message(message.chat.id, f"💱 **Результат:**\n`{amount} {base}` = `{round(final_amount, 2)} {target}`", parse_mode='Markdown')
    except ValueError:
        safe_send_message(message.chat.id, "❌ Введите корректное число.")
    except Exception as e:
        logger.error(f"Currency conversion error: {e}")
        safe_send_message(message.chat.id, f"Ошибка: {e}")

@bot.message_handler(commands=['add_money'])
def admin_add_money(message):
    # Проверка на админа
    if str(message.chat.id) != ADMIN_ID:
        logger.warning(f"Unauthorized add_money attempt from {message.chat.id}")
        return 

    try:
        # Разбираем сообщение: /add_money 123456789 10000
        parts = message.text.split()
        if len(parts) != 3:
            safe_send_message(message.chat.id, "⚠️ Формат: `/add_money <ID_ЮЗЕРА> <СУММА>`", parse_mode='Markdown')
            return

        target_id = int(parts[1])
        amount = float(parts[2])

        # Используем твою функцию обновления баланса
        update_usdt_balance(target_id, amount, "Admin Grant")

        safe_send_message(message.chat.id, f"✅ Успешно!\n👤 Юзер: `{target_id}`\n💰 Начислено: **${amount} USDT**", parse_mode='Markdown')

        # Уведомляем юзера (по желанию)
        try:
            safe_send_message(target_id, f"🎁 **Бонус от администратора!**\nВаш баланс пополнен на **${amount} USDT**", parse_mode='Markdown')
        except: pass

    except ValueError:
        safe_send_message(message.chat.id, "❌ Ошибка. ID и Сумма должны быть числами.")
    except Exception as e:
        logger.error(f"Admin add_money error: {e}")
        safe_send_message(message.chat.id, f"❌ Ошибка: {e}")
