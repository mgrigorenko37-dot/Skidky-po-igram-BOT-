ИСПРАВЛЕНИЕ 1: Починить сбор прибыли (Timezone Fix)
Замени функцию update_mining_buffer на эту версию (используем UTC, чтобы не было ошибок с часовыми поясами):

def update_mining_buffer(user_id):
    try:
        with sqlite3.connect(DB_FILE) as conn:
            cursor = conn.cursor()
            row = cursor.execute("SELECT buffer_usdt, last_collect_date FROM mining_users WHERE user_id = ?", (user_id,)).fetchone()
            
            if not row:
                cursor.execute("INSERT INTO mining_users (user_id, reg_date, balance_usdt, buffer_usdt) VALUES (?, ?, 0.0, 0.0)", 
                             (user_id, datetime.utcnow().strftime('%Y-%m-%d')))
                conn.commit()
                return 0.0, 0.0, 0
            
            buffer, last_date = row
            now = datetime.utcnow()
            income_per_day = calculate_farm_income(user_id)
            
            # Лимит = доход за 2 дня
            max_buffer = round(income_per_day * 2, 6)
            if max_buffer < 0.001: max_buffer = 0.001
            
            if not last_date:
                cursor.execute("UPDATE mining_users SET last_collect_date = ? WHERE user_id = ?", (now.isoformat(), user_id))
                conn.commit()
                return buffer, max_buffer, 0
                
            try:
                last_dt = datetime.fromisoformat(last_date)
            except:
                last_dt = now

            # Считаем разницу в секундах
            seconds_passed = (now - last_dt).total_seconds()
            if seconds_passed < 0: seconds_passed = 0
            
            income_per_second = income_per_day / 86400.0
            new_accrual = seconds_passed * income_per_second
            
            final_buffer = min(buffer + new_accrual, max_buffer)
            
            cursor.execute("UPDATE mining_users SET buffer_usdt = ?, last_collect_date = ? WHERE user_id = ?", 
                          (final_buffer, now.isoformat(), user_id))
            conn.commit()
            
            percent = int((final_buffer / max_buffer * 100)) if max_buffer > 0 else 0
            if percent > 100: percent = 100
            
            return round(final_buffer, 6), max_buffer, percent
    except Exception as e:
        logger.error(f"Buffer error: {e}")
        return 0, 0, 0

Вот правильный кусок кода для вставки внутрь обработчика buy_box:

    # ... (после получения part_type и rarity)

    with sqlite3.connect(DB_FILE) as conn:

        auto_installed = False
        if old_part:

            if new_power > old_power:
                # Новая лучше! Снимаем старую

        else:
            # Слота вообще не было, ставим новую
            cursor.execute("INSERT INTO mining_inventory (user_id, part_type, rarity, tier, is_installed) VALUES (?, ?, ?, ?, 1)", (user_id, part_type, rarity, tier))
            new_id = cursor.lastrowid
            cursor.execute("INSERT INTO mining_farm (user_id, slot_name, inventory_item_id, install_date) VALUES (?, ?, ?, ?)", (user_id, part_type, new_id, str(datetime.now())))

        if not auto_installed:
            # Если новая хуже, просто кладем в инвентарь
            cursor.execute("INSERT INTO mining_inventory (user_id, part_type, rarity, tier, is_installed) VALUES (?, ?, ?, ?, 0)", (user_id, part_type, rarity, tier))
        
        conn.commit()
